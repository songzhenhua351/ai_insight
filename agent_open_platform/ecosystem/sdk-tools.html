代码示例

基础使用示例

创建和调用智能体 
// Java示例
public class AgentExample {
    public static void main(String[] args) {
        AgentClient client = AgentClient.builder()
            .apiKey("your-api-key")
            .build();
        
        // 创建智能体
        Agent agent = client.createAgent()
            .name("客服助手")
            .description("智能客服机器人")
            .model("gpt-4")
            .build();
        
        // 发送消息
        ChatResponse response = agent.chat()
            .message("你好，我需要帮助")
            .send();
        
        System.out.println(response.getContent());
    }
}

高级功能示例

工具调用和流式响应 
# Python示例
import asyncio
from agent_platform import AgentClient, Tool

async def main():
    client = AgentClient(api_key="your-api-key")
    
    # 定义工具
    weather_tool = Tool(
        name="get_weather",
        description="获取天气信息",
        parameters={
            "city": {"type": "string", "description": "城市名称"}
        }
    )
    
    # 创建带工具的智能体
    agent = await client.create_agent(
        name="天气助手",
        tools=[weather_tool],
        streaming=True
    )
    
    # 流式对话
    async for chunk in agent.chat_stream("北京今天天气怎么样？"):
        print(chunk.content, end="", flush=True)

if __name__ == "__main__":
    asyncio.run(main())

系统集成示例

Spring Boot集成 
// Spring Boot配置
@Configuration
@EnableConfigurationProperties(AgentProperties.class)
public class AgentConfig {
    
    @Bean
    public AgentClient agentClient(AgentProperties properties) {
        return AgentClient.builder()
            .apiKey(properties.getApiKey())
            .endpoint(properties.getEndpoint())
            .timeout(Duration.ofSeconds(30))
            .build();
    }
}

@RestController
@RequestMapping("/api/chat")
public class ChatController {
    
    @Autowired
    private AgentClient agentClient;
    
    @PostMapping("/message")
    public ResponseEntity<ChatResponse> sendMessage(
        @RequestBody ChatRequest request) {
        
        Agent agent = agentClient.getAgent(request.getAgentId());
        ChatResponse response = agent.chat()
            .message(request.getMessage())
            .userId(request.getUserId())
            .send();
            
        return ResponseEntity.ok(response);
    }
}
